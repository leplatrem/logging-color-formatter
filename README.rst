A colored logging formatter.

Installation
------------

::

    pip install logging-color-formatter


Usage
-----

.. code-block:: ini

    [loggers]
    keys = root

    [handlers]
    keys = console

    [formatters]
    keys = generic

    [logger_root]
    level = DEBUG
    handlers = console

    [handler_console]
    class = StreamHandler
    args = (sys.stdout,)
    level = NOTSET
    formatter = generic

    [formatters]
    keys = color

    [formatter_color]
    class = logging_color_formatter.ColorFormatter
